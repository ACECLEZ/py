prinical_amount = int(input("Principal amount (P):"))
annual_interest_rate = int(input("Annual interest rate (r, in percentage):"))
number_of_times_interest_is_compounded_per_year = int(input("Number of times interest is compounded per year (n):"))
number_of_years = int(input("Number of years (t):"))

final_amount = prinical_amount * ((1 + annual_interest_rate / 100 / number_of_times_interest_is_compounded_per_year) ** (number_of_times_interest_is_compounded_per_year * number_of_years))
print("{:.2f}".format(final_amount) )
